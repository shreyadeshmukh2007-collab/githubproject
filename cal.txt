using namespace std;
class sum
{

    char o;

        cout<<"enter input= ";

    }

    {
        return a+b;
    }
    int subtraction()
    {
        return a-b;
    }
    int mul()
    {
        return a*b;
    }
    int division()
    {
        return a/b;
    }
    int mod()
    {
     return a%b;
    }
    void calculator()
    {
        if(o == ('+'))
        display (addition());
        else if(o == ('-'))
        display (subtraction());

        display (mul());
        else if(o == ('%'))
        display (mod());
        else if(o == ('/'))
        display (division());
        else 
        cout<< "no";                                             
    }
    void display(int result)
    {
        cout<<"result="<<result;
    }
